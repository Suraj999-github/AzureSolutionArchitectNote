Strings — 10 problems
1Check if a string is palindrome▼
Compare characters from both ends moving inward. A palindrome reads the same forward and backward (e.g. "racecar"). No built-in reverse needed.
static bool IsPalindrome(string s) {
    int l = 0, r = s.Length - 1;
    while (l < r) {
        if (s[l] != s[r]) return false;
        l++; r--;
    }
    return true;
}
// IsPalindrome("racecar") → true
// IsPalindrome("hello")   → false
2Find first non-repeating character▼
Pass 1: count frequency of every character into a Dictionary. Pass 2: return the first character whose count is exactly 1.
static char FirstNonRepeating(string s) {
    var freq = new Dictionary<char, int>();
    foreach (var c in s)
        freq[c] = freq.GetValueOrDefault(c) + 1;
    foreach (var c in s)
        if (freq[c] == 1) return c;
    return '\0';
}
// FirstNonRepeating("aabbcd") → 'c'
3Count occurrences of each character▼
Iterate through the string, incrementing the count for each character in a Dictionary. Return the map for reporting.
static Dictionary<char, int> CharCount(string s) {
    var map = new Dictionary<char, int>();
    foreach (var c in s)
        map[c] = map.GetValueOrDefault(c) + 1;
    return map;
}
// CharCount("hello") → {h:1, e:1, l:2, o:1}
4Remove duplicate characters from string▼
Use a HashSet to track seen characters. Only append a character to the StringBuilder result if it hasn't been seen before.
static string RemoveDuplicates(string s) {
    var seen = new HashSet<char>();
    var sb = new System.Text.StringBuilder();
    foreach (var c in s)
        if (seen.Add(c)) sb.Append(c);
    return sb.ToString();
}
// RemoveDuplicates("programming") → "progamin"
5Check if two strings are anagrams▼
Two strings are anagrams if they have identical character frequencies. Increment counts for string A and decrement for B; any non-zero entry means not an anagram.
static bool AreAnagrams(string a, string b) {
    if (a.Length != b.Length) return false;
    int[] freq = new int[256];
    foreach (var c in a) freq[c]++;
    foreach (var c in b) freq[c]--;
    foreach (var f in freq)
        if (f != 0) return false;
    return true;
}
// AreAnagrams("listen", "silent") → true
6String compression (aaabb → a3b2)▼
Walk through the string counting consecutive repeated characters. Append the character and its count to the result. Only return compressed form if it is shorter.
static string Compress(string s) {
    var sb = new System.Text.StringBuilder();
    int i = 0;
    while (i < s.Length) {
        char c = s[i]; int count = 1;
        while (i + count < s.Length && s[i + count] == c)
            count++;
        sb.Append(c); sb.Append(count);
        i += count;
    }
    var result = sb.ToString();
    return result.Length < s.Length ? result : s;
}
// Compress("aaabb") → "a3b2"
// Compress("abc")   → "abc" (not shorter)
7Find longest substring without repeating characters▼
Sliding window: expand right pointer. When a duplicate appears, shrink from the left until the window has no repeats. Track max window size.
static int LongestUnique(string s) {
    var seen = new HashSet<char>();
    int l = 0, max = 0;
    for (int r = 0; r < s.Length; r++) {
        while (seen.Contains(s[r]))
            seen.Remove(s[l++]);
        seen.Add(s[r]);
        max = Math.Max(max, r - l + 1);
    }
    return max;
}
// LongestUnique("abcabcbb") → 3 ("abc")
8Convert string to title case without built-in▼
Split by space, capitalize the first character of each word using ASCII arithmetic (subtract 32 to uppercase), lowercase the rest, then join.
static string ToTitleCase(string s) {
    var words = s.Split(' ');
    for (int i = 0; i < words.Length; i++) {
        if (words[i].Length == 0) continue;
        char first = (char)(words[i][0] >= 'a'
            ? words[i][0] - 32 : words[i][0]);
        words[i] = first + words[i].Substring(1).ToLower();
    }
    return string.Join(" ", words);
}
// ToTitleCase("hello world") → "Hello World"
9Implement your own StringBuilder-like class▼
Back the class with a List<char> for O(1) amortized appends. Expose Append (chainable), Length, and ToString. Avoids repeated string concatenation overhead.
class MyStringBuilder {
    private List<char> _chars = new();

    public MyStringBuilder Append(string s) {
        foreach (var c in s) _chars.Add(c);
        return this;
    }
    public MyStringBuilder Append(char c) {
        _chars.Add(c); return this;
    }
    public int Length => _chars.Count;
    public override string ToString() =>
        new string(_chars.ToArray());
}

// → "Hello!"
10Reverse a string without built-in methods▼
Convert to char array, then use two pointers from each end, swapping characters until they meet in the middle. O(n) time, O(n) space.
static string Reverse(string s) {
    char[] arr = s.ToCharArray();
    int l = 0, r = arr.Length - 1;
    while (l < r) {
        char tmp = arr[l];
        arr[l++] = arr[r];
        arr[r--] = tmp;
    }
    return new string(arr);
}
// Reverse("hello") → "olleh"
Arrays — 6 problems
1Find second largest element▼
Single-pass tracking of both first and second largest. When a new max is found, push current max down to second. Update second directly if not a new overall max.
static int SecondLargest(int[] arr) {
    int first = int.MinValue, second = int.MinValue;
    foreach (var n in arr) {
        if (n > first) { second = first; first = n; }
        else if (n > second && n != first) second = n;
    }
    return second;
}
// SecondLargest([3,1,4,1,5,9,2,6]) → 6
2Move all zeros to end▼
Use a write pointer. Copy every non-zero element to the front in order, then fill the remaining positions with zeros. Preserves relative order of non-zero elements.
static void MoveZerosToEnd(int[] arr) {
    int pos = 0;
    foreach (var n in arr)
        if (n != 0) arr[pos++] = n;
    while (pos < arr.Length)
        arr[pos++] = 0;
}
// [0,1,0,3,12] → [1,3,12,0,0]
3Find missing number in array (1 to n)▼
The expected sum of 1 through n is n*(n+1)/2. Subtracting the actual array sum gives the missing number. O(n) time, O(1) space.
static int FindMissing(int[] arr, int n) {
    int expected = n * (n + 1) / 2;
    int actual = arr.Sum();
    return expected - actual;
}
// FindMissing([1,2,4,5,6], 6) → 3
4Find duplicates in array▼
HashSet.Add returns false if the element already exists. Collect every element that fails to add — those are the duplicates.
static List<int> FindDuplicates(int[] arr) {
    var seen = new HashSet<int>();
    var dups = new List<int>();
    foreach (var n in arr)
        if (!seen.Add(n)) dups.Add(n);
    return dups;
}
// FindDuplicates([1,2,3,2,4,3]) → [2, 3]
5Merge two sorted arrays▼
Use three pointers: i for array A, j for array B, k for output. Always pick the smaller current element. Copy remaining elements after one array is exhausted.
static int[] MergeSorted(int[] a, int[] b) {
    int[] result = new int[a.Length + b.Length];
    int i = 0, j = 0, k = 0;
    while (i < a.Length && j < b.Length)
        result[k++] = (a[i] <= b[j]) ? a[i++] : b[j++];
    while (i < a.Length) result[k++] = a[i++];
    while (j < b.Length) result[k++] = b[j++];
    return result;
}
// MergeSorted([1,3,5],[2,4,6]) → [1,2,3,4,5,6]
6Sort array without built-in sort▼
Bubble Sort: repeatedly pass through the array, swapping adjacent elements that are out of order. Each pass moves the largest unsorted element to its correct position.
static void BubbleSort(int[] arr) {
    int n = arr.Length;
    for (int i = 0; i < n - 1; i++)
        for (int j = 0; j < n - i - 1; j++)
            if (arr[j] > arr[j + 1]) {
                int tmp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = tmp;
            }
}
// BubbleSort([5,3,1,4,2]) → [1,2,3,4,5]

5Pagination using LINQ▼
Skip the items from previous pages using (pageNumber-1)*pageSize, then Take the next pageSize items. This is the standard paging pattern used in APIs and data grids.
static IEnumerable<T> Paginate<T>(
    IEnumerable<T> source,
    int pageNumber,
    int pageSize) {
    return source
        .Skip((pageNumber - 1) * pageSize)
        .Take(pageSize);
}

Recursion — 7 problems
1Factorial using recursion▼
Base case: 0! = 1. Recursive case: n! = n × (n-1)!. Each call reduces n by 1 until the base case. Classic demonstration of call stack unwinding.
static long Factorial(int n) {
    if (n <= 0) return 1;
    return n * Factorial(n - 1);
}
// Factorial(5) → 120
// Stack: 5 × 4 × 3 × 2 × 1 × 1
2Fibonacci series▼
F(0)=0, F(1)=1, F(n)=F(n-1)+F(n-2). Add memoization (Dictionary cache) to reduce time from O(2^n) to O(n) by avoiding duplicate sub-calls.
static int Fib(int n,
    Dictionary<int, int> memo = null) {
    memo ??= new();
    if (n <= 1) return n;
    if (memo.ContainsKey(n)) return memo[n];
    memo[n] = Fib(n - 1, memo) + Fib(n - 2, memo);
    return memo[n];
}
// Fib(10) → 55
3Reverse string using recursion▼
Take the last character, prepend the reverse of the remaining string. Base case: 0 or 1 characters — nothing to reverse.
static string ReverseRec(string s) {
    if (s.Length <= 1) return s;
    return ReverseRec(s.Substring(1)) + s[0];
}
// ReverseRec("hello") → "olleh"
// hello → ello → llo → lo → o (unwind adds h,e,l,l)
4Sum of digits▼
Extract the last digit with n%10, add it to the recursive result of n/10. Base case: n==0 returns 0.
static int SumDigits(int n) {
    if (n == 0) return 0;
    return (n % 10) + SumDigits(n / 10);
}
// SumDigits(1234) → 10  (1+2+3+4)
5Binary search using recursion▼
Compare target with the middle element. Recurse into the left half if smaller, right half if larger. Return -1 when left exceeds right (not found).
static int BinarySearch(int[] arr,
    int target, int l, int r) {
    if (l > r) return -1;
    int mid = (l + r) / 2;
    if (arr[mid] == target) return mid;
    if (arr[mid] > target)
        return BinarySearch(arr, target, l, mid - 1);
    return BinarySearch(arr, target, mid + 1, r);
}
// BinarySearch([1,3,5,7,9], 7, 0, 4) → index 3
6Tower of Hanoi▼
Move n-1 disks to the auxiliary peg, move the largest disk to the target, then move n-1 disks from auxiliary to target. Requires 2^n - 1 moves.
static void Hanoi(int n,
    char from, char to, char aux) {
    if (n == 0) return;
    Hanoi(n - 1, from, aux, to);
    Console.WriteLine(
        $"Move disk {n}: {from} → {to}");
    Hanoi(n - 1, aux, to, from);
}
// Hanoi(3, 'A', 'C', 'B') → 7 moves
7Check palindrome recursively▼
Compare first and last characters. If they match, recurse on the inner substring. Base case: 0 or 1 characters always return true.
static bool IsPalindromeRec(string s) {
    if (s.Length <= 1) return true;
    if (s[0] != s[s.Length - 1]) return false;
    return IsPalindromeRec(
        s.Substring(1, s.Length - 2));
}
// IsPalindromeRec("racecar") → true

6Deadlock simulation▼
Deadlock: Thread A holds Lock1 and waits for Lock2; Thread B holds Lock2 and waits for Lock1. Prevention: always acquire locks in a consistent global order across all threads.
object lockA = new(), lockB = new();

// Thread 1: A → B
new Thread(() => {
    lock(lockA) {
        Thread.Sleep(50); // give T2 time to grab B
        lock(lockB) { Console.WriteLine("T1 done"); }
    }
}).Start();

// Thread 2: B → A  ← deadlock with T1!
new Thread(() => {
    lock(lockB) {
        Thread.Sleep(50);
        lock(lockA) { Console.WriteLine("T2 done"); }
    }
}).Start();
7CancellationToken example▼
CancellationTokenSource controls cancellation. Pass the token to long-running operations and call ThrowIfCancellationRequested() at checkpoints. Graceful shutdown without aborting threads.
static async Task RunWithCancellation() {
    using var cts =
        new CancellationTokenSource();
    cts.CancelAfter(TimeSpan.FromSeconds(3));

    try {
        while (true) {
            cts.Token.ThrowIfCancellationRequested();
            Console.WriteLine("Processing...");
            await Task.Delay(1000, cts.Token);
        }
    } catch (OperationCanceledException) {
        Console.WriteLine("Task cancelled gracefully.");
    }
}
8File processing using async▼

static async Task ProcessFileAsync(string inputPath) {

        await File.ReadAllTextAsync(inputPath);

    var processed = content
        .Split('\n')

}
Data Structures — 6 problems
1Implement Stack using array▼
A stack is LIFO (Last In, First Out). Use a top index starting at -1. Push increments and places value; Pop returns and decrements; Peek reads without removing.
class MyStack {
    private int[] _arr;
    private int _top = -1;

    public MyStack(int capacity)
        => _arr = new int[capacity];

    public void Push(int val) {
        if (_top == _arr.Length - 1)
            throw new InvalidOperationException("Stack full");
        _arr[++_top] = val;
    }
    public int Pop() {
        if (IsEmpty) throw new InvalidOperationException();
        return _arr[_top--];
    }
    public int Peek() => _arr[_top];
    public bool IsEmpty => _top == -1;
}
2Implement Queue using two stacks▼
Stack1 receives all enqueues. On dequeue, if Stack2 is empty, pour everything from Stack1 into Stack2 (reversing the order). Then pop from Stack2 for FIFO behavior.
class QueueFromStacks {
    private Stack<int> s1 = new();
    private Stack<int> s2 = new();

    public void Enqueue(int val) => s1.Push(val);

    public int Dequeue() {
        if (s2.Count == 0)
            while (s1.Count > 0)
                s2.Push(s1.Pop()); // transfer reverses order
        return s2.Pop();
    }
    public bool IsEmpty => s1.Count == 0 && s2.Count == 0;
}
// Enqueue(1,2,3) then Dequeue() → 1 (FIFO)
3Reverse a linked list▼
Iteratively walk with three pointers: prev (null initially), curr, and next. At each node, flip the Next pointer, then advance all three pointers. Return prev as the new head.
class Node { public int Val; public Node Next; }

static Node Reverse(Node head) {
    Node prev = null, curr = head;
    while (curr != null) {
        var next = curr.Next; // save next
        curr.Next = prev;     // flip link
        prev = curr;          // advance prev
        curr = next;          // advance curr
    }
    return prev; // prev is new head
}
// 1→2→3→null  becomes  3→2→1→null
4Detect loop in linked list▼
Floyd's cycle detection (tortoise and hare): slow pointer moves 1 step, fast pointer moves 2. If they ever point to the same node, there is a cycle. O(n) time, O(1) space.
static bool HasCycle(Node head) {
    Node slow = head, fast = head;
    while (fast != null && fast.Next != null) {
        slow = slow.Next;        // 1 step
        fast = fast.Next.Next;   // 2 steps
        if (slow == fast) return true; // cycle!
    }
    return false;
}
// Works in O(n) time and O(1) space
5Binary search tree insertion▼
BST property: left subtree has smaller values, right has larger. Recursively navigate left or right and insert at the null position where you stop. In-order traversal gives sorted output.
class BstNode {
    public int Val;
    public BstNode Left, Right;
    public BstNode(int v) => Val = v;
}

static BstNode Insert(BstNode root, int val) {
    if (root == null) return new BstNode(val);
    if (val < root.Val)
        root.Left = Insert(root.Left, val);
    else if (val > root.Val)
        root.Right = Insert(root.Right, val);
    return root;
}
// Insert sequence: 5,3,7,1 builds balanced BST
6Graph traversal (BFS & DFS)▼
BFS (Breadth-First) uses a Queue — explores level by level, good for shortest path. DFS (Depth-First) uses a Stack or recursion — explores as deep as possible, good for cycle detection. Both need a visited set to avoid revisiting nodes.
var graph = new Dictionary<int, List<int>>();

// BFS — level by level
void Bfs(int start) {
    var visited = new HashSet<int>();
    var queue = new Queue<int>();
    queue.Enqueue(start); visited.Add(start);
    while (queue.Count > 0) {
        var node = queue.Dequeue();
        Console.Write(node + " ");
        foreach (var n in graph[node])
            if (visited.Add(n)) queue.Enqueue(n);
    }
}

// DFS — depth first (replace Queue with Stack,
//        Dequeue→Pop, Enqueue→Push)
